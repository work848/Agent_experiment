def mask_user_emails(users: list) -> 
list:
    for user in users:
        if user.email:
            parts = 
user.email.split('@')
            if len(parts) == 2:
                local, domain = parts
                masked_local = 
local[0] + '*' * (len(local) - 1) if 
len(local) > 1 else '*'
                user.email = 
f"{masked_local}@{domain}"
    return users
